    const len = 4;
 
    for (let i = 0; i < len; i++) {
        for (let j = 0; j < len; j++) {
            if (arr[j] > arr[j + 1]) {
             //  [arr[j], arr[j + 1]] = [arr[j + 1], arr[j]];

        }

};
